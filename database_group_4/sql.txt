create schema if not exists marketplacedb;

create table if not exists marketplacedb.product(
    pid integer NOT NULL, primary key(pid),
    name varchar(64),
    price integer,
    description varchar(255));

create table if not exists marketplacedb.reviews(
    rid integer primary key,
    comment_date date,
    rating integer, content varchar(500),
    pid integer,
    foreign key(pid) references marketplacedb.product(pid));

create table if not exists marketplacedb.user(
    uid integer, primary key (uid),
    first_name varchar(64), last_name varchar(64),
    userName varchar(64),
    UserpassWord varchar(64));
